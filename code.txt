@router.get("/customer-sales-summary", summary="Customer sales summary (badly written SP)")
def get_customer_sales_summary(
    name_filter: str = Query(default=None, description="Optional name substring filter"),
    page_number: int = Query(default=1, ge=1, description="1-based page number"),
    page_size: int = Query(default=100, ge=1, le=1000, description="Rows per page"),
):
    """
    Calls retail.usp_GetCustomerSalesSummary — an intentionally
    poorly-written stored procedure for performance-tuning demos.
    """
    try:
        conn = get_connection()
        cursor = conn.cursor()

        logger.info(
            "▶ Executing retail.usp_GetCustomerSalesSummary (name_filter=%s, page_number=%d, page_size=%d)",
            name_filter,
            page_number,
            page_size,
        )
        start = time.perf_counter()

        cursor.execute(
            "EXEC retail.usp_GetCustomerSalesSummary @NameFilter = ?, @PageNumber = ?, @PageSize = ?",
            name_filter,
            page_number,
            page_size,
        )
        columns = [col[0] for col in cursor.description]
        rows = [dict(zip(columns, row)) for row in cursor.fetchall()]

        total_rows = int(rows[0].get("total_rows", 0)) if rows else 0
        total_pages = math.ceil(total_rows / page_size) if total_rows > 0 else 0
        items = []
        for row in rows:
            clean_row = dict(row)
            clean_row.pop("total_rows", None)
            items.append(clean_row)

        elapsed = time.perf_counter() - start
        logger.info(
            "✔ usp_GetCustomerSalesSummary completed — rows=%d total_rows=%d page=%d/%d in %.2f seconds (%.1f ms)",
            len(items),
            total_rows,
            page_number,
            total_pages,
            elapsed,
            elapsed * 1000,
        )

        cursor.close()
        conn.close()
        return {
            "items": items,
            "page_number": page_number,
            "page_size": page_size,
            "total_rows": total_rows,
            "total_pages": total_pages,
        }
    except Exception as exc:
        raise HTTPException(status_code=500, detail=str(exc)) from exc
